156 /// @brief Insert a node in the branch. All the counters of the
157 // nodes must be incremented from the insertion point to the root.\n
158 // Temporally a branch can have 4 nodes after a insertion. In this

162 /// @param [in] Cod : code of the pointer where connect the node

165 // they reflect the number of nodes after the insertion, not
166 /// before

173 
174  //Preparacion del nodo a insertar

176  P1->init() ;
177 
178  switch( Cod)
179  { case 0: PBlack->left = P1 ;
180  P1->up = PBlack;
181  break ;
182 
183  case 1: PBlack->right = P1 ;
184  P1->up = PBlack;
185  break ;
186 
187  case 2: PBlack->left->left = P1 ;
188  P1->up = PBlack->left ;
189  //PBlack->left->N++ ;
190  break ;
191 
192  case 3: P1->left = PBlack->left ;

194  PBlack->left = P1 ;
195  P1->N = 2 ;
196  P1->left->N = 1 ;
197  P1->up = PBlack ;

395 //---------------------------------------------------------------------------
396 // function : reverse_cake
397 /// @brief Implements the reverse cake mechanism between the
398 /// Rup and Rinf branches
399 /// @param [in] Rup : Upper branch
400 /// @param [in] RInf : lower branch
401 /// @param [in] CodS : Code of the pointer of Rup which connect the
402 /// two branches
403 /// @return Estability of the branch (true : stable,false : Rup is
404 /// empty and the branch is not stable)
405 //---------------------------------------------------------------------------
406 bool branch::reverse_cake (branch &Rup, branch &RInf,unsigned CodS)
407 { //------------------------------- Inicio --------------------------------
408  unsigned Nup = Rup.n_nodes() ;
409  #if __DEBUG_CNTREE != 0
410  assert ( not ( CodS > 5 or ( CodS < 2 and Nup > 2)) );
411  #endif
412  //
413  // branch Superior
414  // Si Cod > 1 indica que tenemos el nodo listo para el descenso
415  // Si Cod < 2 la branch no puede tener mas de 2 nodos
416  // Si Cod < 2 y hay mas de un nodo en la rama hay que hacer una rotacion
417  // para tener el nodo listo para el descenso
418  //
419  if ( CodS < 2 and Nup > 1 )
420  { if ( CodS == 0 )
421  { Rup.rotate_left_aligned() ;
422  CodS = 2 ;
423  }
424  else
425  { Rup.rotate_right_aligned() ;
426  CodS = 5 ;
427  };
428  };
429  //
430  // Presuponemos que RInf esta vacia, por lo que no nos vale para nada.
431  // Hay que buscar la rama que cuelga del mismo nodo que RInf.
432  // Si dicha rama tiene mas de 1 nodo, mediante una rotacion inserta un
433  // nodo en RInf y el sistema queda estable.
434  // Si solo tiene 1 nodo , descendemos el nod de la rama superior.
435  // Si dicha rama tiene mas de 1 nodo , el sistema est√° estable.
436  // Si solo tiene un nodo , queda vac√≠a y la inestabilidad se propaga
437  // hacia la rama superior
438  //
439  // El nuevo codigo es el mismo que el anterior con el ultimo bit complementado
440  unsigned CodS2 = ( CodS & 6) | ((~CodS) & 1);
441  base_node ** PPInf2 = Rup.pointer_Cod( CodS2);
442  branch RInf2( PPInf2) ;
443  unsigned NInf2 = RInf2.n_nodes() ;
444 
445  // Si RInf2 tiene mas de 1 nodo se arregla con una rotaci√≥n
446  if ( NInf2 > 1)
447  { base_node * PBlack = * RInf2.ppblack ;
448  base_node ** PPup2 = Rup.pointer_father ( PBlack->up);
449  unsigned color2 = PBlack->up->color ;
450  if ( (CodS2 & 1 ) == 0 )
451  { if ( PBlack->right != NULL and PBlack->right->color == red)
452  base_node::rotate_right_not_aligned ( PPup2);
453  else base_node::rotate_right_aligned ( PPup2);
454  }
455  else
456  { if ( PBlack->left != NULL and PBlack->left->color == red)
457  base_node::rotate_left_not_aligned ( PPup2) ;
458  else base_node::rotate_left_aligned( PPup2) ;
459  };
460  PBlack = *PPup2 ;
461  PBlack->color= color2 ;
462  PBlack->left->color = black ;
463  PBlack->right->color = black ;
464  return true ;
465  };
466  //
467  // El nodo de la rama superior desciende. Si la rama superior tiene
468  // mas de 1 nodo, al finalizar el descenso , queda estable.
469  // Si solo ten√≠a 1 nodo , queda vac√≠a y por lo tanto inestable
470  // En este punto sabemos con certeza que RInf2 tiene 1 solo nodo,
471  // por lo que el descenso es trivial
472  //
473  base_node * PAux = *RInf2.ppblack ;
474  PAux->up->color = black ;
475  PAux->color = red ;
476  return ( Nup >1);
477 };
