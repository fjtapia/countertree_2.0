259 // function : allocate
260 /// @brief find the first position free and allocate an element
261 /// @param [in] IsFull : Indicate if the heap is full after the allocation
262 /// @return The position of the element allocated. If full return MAX64
263 //------------------------------------------------------------------------
264 uint64_t allocate ( void) NOEXCEPT
265 { //------------------------- begin -----------------------------
266  if ( is_full() ) return MAX64 ;
267  //--------------------------------------------------------------------
268  // If the only operations which modify the heap a re allocate and deallocate
269  // the cCursor can guarantee what the free positions in the heap are from the
270  // Cursor position forward
271  //---------------------------------------------------------------------------
272  #if __DEBUG_CNTREE != 0
273  assert ( Cursor < NElemMax) ;
274  #endif
275  bool Invalid = false ;
276  if ( (Invalid = read_bit(Cursor)) )
277  { uint64_t CursorWord = Cursor >>6 ;
278  uint64_t Aux = Ptr[ CursorWord ];
279  if ( not (Invalid = (Aux == MAX64)) )
280  Cursor = (CursorWord <<6)+ LS1B_64 ( ~Aux);
281  };
282  uint64_t Level ;
283  if ( Invalid )
284  { for ( Cursor = 0, Level = 0 ; Level < NLevel ;++Level)
285  { Cursor=(Cursor<<6)+ LS1B_64(~ L[Level][Cursor]);
286  };
287  };
288  //---------------------------------------------------------------------
289  // set to one the bit
290  //---------------------------------------------------------------------
291  uint64_t N1 = Cursor , N2 = 0 ;
292  Level = NLevel;
293  uint64_t * PAux = NULL ;
294  do
295  { N2 = N1 & 63 ;
296  PAux = &(L[--Level][N1>>=6]);
297  #if __DEBUG_CNTREE != 0
298  assert ( not ReadBit64(*PAux , N2)) ;
299  #endif
300  SetBit64 ( *PAux, N2);
301  #if __DEBUG_CNTREE != 0
302  assert ( ReadBit64(*PAux , N2)) ;
303  #endif
304  } while ( (*PAux == MAX64) and Level != 0) ;
305  NElem++;
306  return ( Cursor++);
307 };

331  #endif
332  } while ( SW and Level != 0) ;
333  if ( Pos < Cursor) Cursor = Pos ;
334  NElem-- ;
335 };
