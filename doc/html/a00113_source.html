558 // function : emplace_internal
559 /// @brief insert an element in a the position specified by the connector
560 /// @param [in] BM : barrier modify needed for to exec the wait_no_readers
561 /// inside the function
562 /// @param [in] C : connector which indicate where must insert the node
563 /// @param [in] args : arguments to build the node to insert
564 /// @return iterator to the element inserted
565 /// @remarks This operation is O ( log(N) ) This operation don't must be used
566 /// by the final users
567 //---------------------------------------------------------------------------
568 template <class ... Args>
569 const_iterator emplace_internal(mtx_write & BM,connector C, Args && ... args)
570 { //------------------ begin-----------------
571  node_t * PAux = st.allocate (1) ;
572  st.construct ( PAux , std::forward<Args> ( args) ... );
573  BM.wait_no_readers() ;
574  st.connect ( PAux , C ) ;
575  return st.get_cit ( PAux) ;
576 };
577 //---------------------------------------------------------------------
578 // function : insert
579 /// @brief insert a value in the cntree_set
580 /// @param [in] val : value to insert.It can ve a value , a reference or an rvalue
581 /// @return pair <iterator,bool> with the iterator to the element inserted or
582 /// or to the element that prevented the insertion, and a bool indication
583 /// if the insertion is done
584 //----------------------------------------------------------------
585 template <class P >
586 mypair insert ( P && Val)
587 { //------------------- begin --------------------------
588  mtx_write BM ( BD);
589  connector C ;
590  node_t *PAux = st.connector_norep ( std::forward<P>(Val), C );
591  bool SW = ( PAux == NULL);
592  const_iterator It = (SW)? emplace_internal(BM,C,std::forward<P>(Val)): st.get_cit (PAux);
593  return mypair ( It, SW);
594 };

597 /// @brief insert a value in the cntree_set
598 /// @param [in] val : value to insert.It can ve a value , a reference or an rvalue
599 /// @param [in] M : function with the format bool( const value_t &).
600 /// if return true the element is iserted , if false not
601 /// @return pair <iterator,bool> with the iterator to the element inserted or
602 /// or to the element that prevented the insertion, and a bool indication
603 /// if the insertion is done.
604 /// When the element is rejected by the function , the itertor of the pair
605 /// is end() and the bool false
606 //----------------------------------------------------------------
607 template <class Function, class P >
608 mypair insert_if (Function && M ,P && Val)
609 { //------------------------begin--------------------------------

612  node_t *PAux = st.connector_norep ( std::forward<P>(Val), C );
613  const_iterator It ;

618  }
619  else It = st.get_cit ( PAux);
620  return mypair ( It, SW);
621 };
