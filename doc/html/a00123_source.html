131 /// @brief create a new row with double size than the previous
132 /// @exception std::bad_alloc when fails the allocation

135 void create_row (void)
136 { //-------------------------begin------------------------------------
137  if ( NRow == Power2Max )throw std::bad_alloc();
138 
139  V[NRow]= NULL ;
140  if (NRow == 0)
141  { void * PAux = AB.allocate ( sizeof (heap32<0,AllocByte>) );
142  if ( PAux== NULL ) throw std::bad_alloc();
143  ::new ((void*)PAux) heap32 < 0,AllocByte>;
144  V[0]= ( basic_heap32<AllocByte> *) PAux ;
145  }
146  else V[NRow] = V[NRow-1]->create_son(AB);
147 
148  ResetBit32 ( Mask , NRow);
149  NMax += V[NRow]->capacity();
150 
151  Ptr[NRow][0] = NULL;
152  Ptr[NRow][0] = AB.allocate ( V[NRow]->NElemMax * SizeElem);
153  if ( Ptr[NRow][0] == NULL )
154  { AB.deallocate ( (uint8_t*)V[NRow],(V[NRow]->SIZEOF()) );
155  V[NRow ] = NULL ;
156  throw std::bad_alloc();
157  };
158 
159  Ptr[NRow][1] = Ptr[NRow][0] + (V[NRow]->NElemMax * SizeElem) ;
160  NRow ++ ;
161 };

172  NMax -= V[NRow]->capacity();
173  AB.deallocate ( (uint8_t*)Ptr[NRow][0], (V[NRow]->NElemMax * SizeElem) );
174  AB.deallocate ((uint8_t*)V[NRow],( V[NRow]->SIZEOF()) );
