150 // function : previous
151 /// @brief Obtain the pnode to the previous node. If don't exist return NULL
152 /// @param [in] none
153 /// @return Pointer to the node.
154 /// @remarks This operation is O ( log N )
155 //---------------------------------------------------------------------------
156 base_node * previous ( void ) NOEXCEPT
157 { //------------- begin ------------------
158  base_node * P = this ;
159 
160  // Primero tratamos de buscarlo hacia abajo. Si no fuera asi , lo
161  // buscamos hacia arriba
162  if ( left != NULL )
163  { P = left ;
164  while ( P->right != NULL ) P = P->right ;
165  }
166  else
167  { // Subir por la dcha todo lo que puedas y luego un salto hacia
168  // arriba a la izda
169  // Si no es posible saltar a la izda devuelve NULL, porque no
170  // tiene nada posterior
171  while ( P->up != NULL and P->up->left == P ) P = P->up ;
172  P =( P->up != NULL and P->up->right != P ) ?NULL :P->up ;
173  };
174  return P ;
175 };

209 //---------------------------------------------------------------------------
210 // function : shift
211 /// @brief Obtain the pnode of the node located several positions
212 /// after or before the node
213 /// @param [in] NDesp : Number of positions to shift. This value can be
214 /// positive or negative
215 /// @return Pointer to the node. If don't exist return NULL
216 /// @remarks This operation is O ( log N )
217 //---------------------------------------------------------------------------
218 base_node * shift ( signed_type NDesp ) NOEXCEPT
219 { //----------------- begin ---------------
220  base_node * P = this ;
221  if ( NDesp == 0 ) return P;
222  while ( NDesp != 0)
223  { if (NDesp > 0 )
224  { // Desplazamiento hacia adelante
225  // Trata de bajar por la dcha. Lo maximo que puede avanzar por
226  // ese camino est√° determinado por el contador del nodo a su dcha.
227  // Si esto no fuera suficiente, emprende el camino hacia el nodo
228  // superior. Si este no existiera quiere decir que no puede
229  // alcanzar ese desplazamiento por lo que devuelve NULL
230  //
231  if ( NDesp > (signed_type)(P->n_right()) )
232  { if ( P->up == NULL ) return NULL ;
233  if ( P->up->right == P ) NDesp += P->n_left() + 1;
234  else NDesp -= P->n_right() + 1;
235  P = P->up ;
236  }
237  else
238  { // Bajada por la derecha
239  P = P->right ;
240  NDesp -= P->n_left() +1 ;
241  };
242  }
243  else
244  { //
245  // Desplazamiento hacia atras
246  // Trata de bajar por la left. Lo m√°ximo que puede retroceder por
247  // ese camino est√° determinado por el contador del nodo a su izda.
248  // Si esto no fuera suficiente, emprende el camino hacia el nodo
249  // superior. Si este no existiera quiere decir que no puede
250  // alcanzar ese desplazamiento por lo que devuelve NULL
251  //
252  if ( -NDesp > (signed_type)(P->n_left()) )
253  { if ( P->up == NULL ) return NULL ;
254  if ( P->up->right == P ) NDesp += P->n_left () + 1;
255  else NDesp -= P->n_right () + 1;
256  P = P->up ;
257  }
258  else
259  { // Bajada por la izda
260  P = P->left ;
261  NDesp += P->n_right() +1 ;
262  };
263  };
264  }; // Fin while
265  return P;
266 };
267 const base_node * shift ( signed_type NDesp ) const NOEXCEPT
268 { return remove_const()->shift ( NDesp);

339 /// @brief Swap two nodes. They can't be contiguous
340 /// @param [in] PP1 : Pointer to the base_node pointer to the first node
341 /// @param [in] PP2 : Pointer to the base_node pointer to the second node
342 /// @return none
343 //---------------------------------------------------------------------------
344 static void swap_node ( base_node ** PP1, base_node ** PP2) NOEXCEPT
345 { //------------------------------- Inicio ---------------------
346  #if __DEBUG_CNTREE != 0
347  assert ( PP1 !=NULL and *PP1 !=NULL and PP2 !=NULL and *PP2 !=NULL);
348  #endif
349  base_node *P1 = *PP1 , *P2 = *PP2 ;
350  *PP1 = P2 ;
351  *PP2 = P1 ;
352 
353  std::swap ( P1->up , P2->up) ;
354  std::swap ( P1->left , P2->left);
355  if ( P1->left != NULL) P1->left->up = P1 ;
356  if ( P2->left != NULL) P2->left->up = P2 ;
357 
358  std::swap ( P1->right , P2->right);
359  if ( P1->right != NULL) P1->right->up = P1 ;
360  if ( P2->right != NULL) P2->right->up = P2 ;
361 
362  unsigned color1 = P1->color ;
363  P1->color= P2->color ;
364  P2->color = color1 ;
365 
366  bitfield_t N1 = P1->N ;
367  P1->N = P2->N ;
368  P2->N = N1 ;
369 };
