421 // function : lower_bound
422 /// @brief find a key in the tree. If there are repeated
423 /// elements, return an const_iterator to the first of them.
424 /// If the key don't exist return an iterator to the first element
425 /// greater than the key
426 /// @param [in] K : key to search
427 /// @return const_iterator to the element. If don't exist end()
428 /// @remarks
429 //---------------------------------------------------------------------------
430 iterator lower_bound ( const key_t & K) NOEXCEPT
431 { //-------------------- begin ---------------------------
432  node_t * P1 = const_cast <node_t*> ( T.get_root() );
433  node_t *P2 = P1;
434  if ( P1 == NULL ) return end() ;
435  iterator C = end() ;
436  bool SWR = false ;
437 
438  // Si encontramos un valor igual, lo apuntamos , y seguimos buscando
439  // por la izda
440  while ( P1 != NULL)
441  { P2 = P1 ;
442  P1 = (node_t *)((SWR = cmpr(fltr(P1->data),K) )? P1->right :P1->left);
443  if ( not SWR)
444  { if ( not cmpr(K , fltr(P2->data)))
445  C = iterator(P2,&T);
446  };
447  };
448  if ( C == end())
449  { if (SWR)
450  { P2 = upgrade(P2->next()) ;
451  C = ( P2 == NULL)? end() :iterator ( P2, &T) ;
452  }
453  else C = iterator( P2 , &T ) ;
454  };
455  return C ;
456 };
