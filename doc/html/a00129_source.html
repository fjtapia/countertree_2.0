142  if ( NRow == Power2Max )throw std::bad_alloc();
143 
144  V[NRow]= NULL ;
145  if (NRow == 0)
146  { void * PAux = AB.allocate ( sizeof (heap64<0,AllocByte> ) );
147  if ( PAux == NULL ) throw std::bad_alloc();
148  ::new ((void*)PAux) heap64 < 0,AllocByte>;
149  V[0]= ( basic_heap64<AllocByte> *) PAux ;
150  }
151  else V[NRow] =V[NRow-1]->create_son(AB);
152 
153  ResetBit64 ( Mask , NRow);
154  NMax += V[NRow]->capacity();
155 
156  Ptr[NRow][0] = NULL;
157  Ptr[NRow][0] = AB.allocate ( V[NRow]->NElemMax * SizeElem);
158  if ( Ptr[NRow][0] == NULL )
159  { AB.deallocate ( (uint8_t*)V[NRow],V[NRow]->SIZEOF() );
160  V[NRow ] = NULL ;
161  throw std::bad_alloc();
162  };
163  Ptr[NRow][1] = Ptr[NRow][0] + (V[NRow]->NElemMax * SizeElem) ;
164  NRow ++ ;
165 };

176  NMax -= V[NRow]->capacity();
